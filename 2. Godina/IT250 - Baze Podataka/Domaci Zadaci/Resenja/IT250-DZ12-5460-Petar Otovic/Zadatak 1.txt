vreme_prenosa_pacijent = (1000 * 120) / 50000 + 0.1

Vreme prenosa Bolest tabele na lokaciju I3:
vreme_prenosa_bolest = (500 * 120) / 50000 + 0.1

ukupno_vreme_prenosa = vreme_prenosa_pacijent + vreme_prenosa_bolest 
ukupno_vreme_prenosa =  2.5 + 1.3
